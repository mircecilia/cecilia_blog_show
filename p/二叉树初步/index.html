def fib_tree(n):
    if n<=1:
        return tree(1)
    else:
        left,right=fib_tree(n-1),fib_tree(n-2)
        return tree(label(left)+label(right),[left,right])

def count_paths(t,total):
    if label(t)==total:
        found = 1
    else:
        found = 0
    return found + sum([count_paths(b,total-label(t)) for b in branches(t)])
